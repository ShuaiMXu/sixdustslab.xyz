// installed · v4.6 · auto-sync on
import { defineSkill } from "@sixdustslab/happyclaw";

export default defineSkill({
  name: "qualify-lead",
  channels: ["whatsapp", "telegram"],
  model: "claude-opus-4.6",

  async run(ctx) {
    const lead = await ctx.extract({
      budget: "number",
      timeline: "weeks",
      needs: "text",
    });
    if (lead.score > 0.75) {
      await ctx.escalate("#sales-hot");
    }
    await ctx.crm.sync(lead);
  },
});
